96 void MaxHeap<K,V>::upHeap(int i) {
97  while (i>1 && a[i].value > a[PARENT(i)].value) { // while pos bigger than parent, keep swapping to upper position

105 void MaxHeap<K,V>::downHeap(int i) {
106  while (LEFT(i) <= size) { // while within heap limits
107  int j = LEFT(i);
108  if (RIGHT(i)<=size && a[RIGHT(i)].value > a[j].value) j = RIGHT(i);
109  if (a[i].value > a[j].value) break;
110  swap(i, j);
111  i = j;
112  }
